357 static void encodeTermination( char *buffer, int total_length, int content_length )
358 {
359  if ( ( total_length >= (CHECKSUM_LENGTH + TERMINATOR_LENGTH) ) && ( total_length >= content_length + (CHECKSUM_LENGTH + TERMINATOR_LENGTH) ) )
360  {
361  // Checksum
362  unsigned char checksum = 0;
363  for ( int i = 0; i < content_length; i++ )
364  {
365  checksum += buffer[i];
366  }
367  // convert checksum to two ascii bytes
368  sprintf(&buffer[content_length], "%02X", checksum);
369  // Message Terminator
370  sprintf(&buffer[content_length + CHECKSUM_LENGTH], "%s","\r\n");
371  }
372 }

384 static void encodeProtocolFloat( float f, char* buff )
385 {
386  int temp1 = abs((int)((f - (int)f) * 100));
387  if ( f < 0 ) buff[0] = '-'; else buff[0] = ' ';
388  sprintf(&buff[1],"%03d.%02d", abs((int)f), temp1);

396 static uint16_t decodeProtocolUint16( char *uint16_string )
397 {
398  uint16_t decoded_uint16 = 0;
399  unsigned int shift_left = 12;
400  for ( int i = 0; i < 4; i++ )
401  {
402  unsigned char digit = uint16_string[i] <= '9' ? uint16_string[i] - '0' : ((uint16_string[i] - 'A') + 10);
403  decoded_uint16 += (((uint16_t)digit) << shift_left);
404  shift_left -= 4;
405  }
406  return decoded_uint16;
407 }

410 static bool verifyChecksum( char *buffer, int content_length )
411 {
412  // Calculate Checksum
413  unsigned char checksum = 0;
414  for ( int i = 0; i < content_length; i++ )
415  {
416  checksum += buffer[i];
417  }
418 
419  // Decode Checksum
420  unsigned char decoded_checksum = decodeUint8( &buffer[content_length] );
421 
422  return ( checksum == decoded_checksum );
423 }

425 static unsigned char decodeUint8( char *checksum )
426 {
427  unsigned char first_digit = checksum[0] <= '9' ? checksum[0] - '0' : ((checksum[0] - 'A') + 10);
428  unsigned char second_digit = checksum[1] <= '9' ? checksum[1] - '0' : ((checksum[1] - 'A') + 10);
429  unsigned char decoded_checksum = (first_digit * 16) + second_digit;
430  return decoded_checksum;
431 }
